The following is our code that completely solves the 10-to-0-by-1-or-2 game, in which two people play a game that starts at 10 with each person taking 1 or 2 on their turn; the winner is the one that gets the total to 0 first: 

LOSE, WIN, TIE, UNDECIDED = "Lose", "Win", "Tie", "Undecided"

def primitive(position):
    "Return (WIN, LOSE, TIE) if POSITION is a primitive (game over), otherwise UNDECIDED"
    return LOSE if position == 0 else UNDECIDED

def generate_moves(position):
    "Return all the moves available from POSITION"
    return (1,) if position == 1 else (1, 2)
    
def do_move(position, move):
    "Return the child position resulting from doing the MOVE from the POSITION
    return position - move

def solve(position):
    "Solve the game (walk the tree starting from POSITION), return (WIN, LOSE, TIE)"
    if primitive(position) != UNDECIDED:
        return primitive(position)
    else:
        values = [solve(do_move(position, move)) for move in generate_moves(position)]
        return WIN if LOSE in values else (TIE if TIE in values else LOSE)
