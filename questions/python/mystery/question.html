def mystery(L): ## L is a list
    DB = {}
    for word in L:
        if word in DB:
            DB[word] = DB[word] + 1
        else:
            DB[word] = 1
    return DB

## f is a function, L is a list
def mystery2(f, L):
    if len(L) == 0:
        return []
    else:
        first_elt = L[0]
        rest_elts = L[1:]
        if f(first_elt):
            return [first_elt] + mystery2(f, rest_elts)
        else:
            return mystery2(f, rest_elts)

def isEven(n):
    return n % 2 == 0

def compose(f, g):
    """Given functions f(x) and g(x), returns
    the function equivalent to f(g(x))"""
    return lambda x: f(g(x))

def mystery3(f, L):
    if (len(L) == 1):
        return L[0]
    else:
        return f(L[0], mystery3(f, L[1:]))

def add(x, y):
    return x + y

def oneplus(n):
    return n + 1

def tentimes(n):
    return n * 10
